let usuarios = ["admin"]; // Lista de usuarios existentes
let correos = ["admin@mail.com"]; // Lista de correos existentes

function registrar() { // Función al presionar "Registrar"

  let usuario = document.getElementById("usuario").value; // Obtiene usuario
  let correo = document.getElementById("correo").value; // Obtiene correo
  let pass = document.getElementById("password").value; // Obtiene contraseña
  let confirmar = document.getElementById("confirmar").value; // Confirmación

  // Verifica si el usuario ya existe
  if (usuarios.includes(usuario)) {
    mostrarModal("Usuario ya existe"); // Muestra error
    return; // Detiene ejecución
  }

  // Verifica si el correo ya está registrado
  if (correos.includes(correo)) {
    mostrarModal("Correo ya registrado"); // Muestra error
    return; // Detiene ejecución
  }

  // Verifica si las contraseñas coinciden
  if (pass !== confirmar) {
    mostrarModal("Contraseñas no coinciden"); // Error
    return; // Detiene ejecución
  }

  // Si todo está correcto
  mostrarModal("Registro exitoso"); // Mensaje éxito
}

function mostrarModal(msg) { // Función mostrar ventana
  document.getElementById("mensaje").innerText = msg; // Cambia texto
